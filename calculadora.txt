        <input type="number" id="tasa" placeholder="Ingresa la tasa de interés" step="any">

        <label for="periodos">Número de períodos (años):</label>
        <input type="number" id="periodos" placeholder="Ingresa el número de períodos" step="any">

        <label for="tipoInteres">Tipo de Interés:</label>
        <select id="tipoInteres">
            <option value="simple">Interés Simple</option>
            <option value="compuesto">Interés Compuesto</option>
        </select>

        <button onclick="calcularInteres()">Calcular</button>

        <h3 id="resultado"></h3>
    </div>

    <script>
        function calcularInteres() {
            const capital = parseFloat(document.getElementById("capital").value);
            const tasa = parseFloat(document.getElementById("tasa").value) / 100;
            const periodos = parseFloat(document.getElementById("periodos").value);
            const tipoInteres = document.getElementById("tipoInteres").value;
            let resultado = 0;

            if (tipoInteres === "simple") {
                resultado = capital * (1 + tasa * periodos);
            } else if (tipoInteres === "compuesto") {
                resultado = capital * Math.pow((1 + tasa), periodos);
            }

            document.getElementById("resultado").textContent = "Resultado: " + resultado.toFixed(2);
        }
    </script>
